// f(x) i bulan programý özyinelemeli ve özyinelemesiz yazýnýz.  
//

//
// 1 ayda cocuk 1 kitap okumuþtur. 
// Bu cocuk ailesine her ay bir önceki ayýn 4 katýndan 
// 2 eksik kitap okuyacaðýna söz vermiþtir. 
// Girilen aya göre cocuðun okuyacaðý kitap sayýsýný ekrana
// yazan programý yazýnýz. 

int kitap_say(int ay)
{
	if(ay==1) return 1; 
	return kitap_say(ay-1)*4-2;
}


int f(int x)
{
	if(x==0) return 2; 
	if(x==1) return 5;
	return f(x-1)*2+f(x-2)/2;
}


int adimsay(int i)
{
	if(i==0) return 4000;
	return adimsay(i-1)*1.1;
}

int main()
{
	int i; 
	for(i=1;i<=12;i++)
	printf("%d ay %d kitap okumalisin!\n",i,kitap_say(i));
}
